908 void update()
909 {
910  float angleRadians = carDirection * M_PI / 180.0f;
911 
912  if (isMovingForward) {
913  carSpeed = min(maxSpeed, carSpeed + acceleration);
914  } else if (isMovingBackward) {
915  carSpeed = max(-maxSpeed / 2, carSpeed - acceleration);
916  } else {
917  // Natural speed decay (simulating friction/drag)
918  if (carSpeed > 0) {
919  carSpeed = max(0.0f, carSpeed - deceleration);
920  } else if (carSpeed < 0) {
921  carSpeed = min(0.0f, carSpeed + deceleration);
922  }
923  }
924 
925  // Calculate dynamic turn rate based on speed
926  float speedFactor = abs(carSpeed) / maxSpeed;
927  float currentTurnSpeed = minTurnSpeed + (maxTurnSpeed - minTurnSpeed) * (1.0f - speedFactor);
928  currentTurnSpeed *= turnSpeedMultiplier;
929 
930  // Smooth turning logic
931  if (isTurningLeft || isTurningRight) {
932  // Gradually increase turn interpolation
933  turnInterpolation = min(1.0f, turnInterpolation + turnAcceleration);
934  } else {
935  // Gradually increase turn interpolation
936  turnInterpolation = max(0.0f, turnInterpolation - turnAcceleration);
937  }
938 
939  // Apply smooth turning
940  if (turnInterpolation > 0) {
941  float turnAmount = currentTurnSpeed * turnInterpolation;
942 
943  if (isTurningLeft) {
944  carDirection += turnAmount;
945  }
946  if (isTurningRight) {
947  carDirection -= turnAmount;
948  }
949 
950 
951  while (carDirection >= 360)
952  carDirection -= 360;
953  while (carDirection < 0)
954  carDirection += 360;
955  }
956 
957  // sin for x (east-west) and cos for z (north-south)
958  carPosition.x += carSpeed * sin(angleRadians);
959  carPosition.z += carSpeed * cos(angleRadians);
960 
961  // Update car heading based on direction
962  if (carDirection >= 315 || carDirection < 45) {
963  carHeading = "N";
964  } else if (carDirection >= 45 && carDirection < 135) {
965  carHeading = "E";
966  } else if (carDirection >= 135 && carDirection < 225) {
967  carHeading = "S";
968  } else {
969  carHeading = "W";
970  }
971 
972  worldCameraOffset = computeRotatedVector(localCameraOffset, carDirection);
973 
974  // State machine for the traffic signals using three variables: NS_Signal,
975  // WE_Signal, and counter.
976  counter += updateInterval; // Add elapsed time in milliseconds
977 
978  const int greenTime = 5000; // 5 seconds
979  const int yellowTime = 1000; // 1 second
980  const int redTime = 6000; // 6 seconds
981  const int totalCycle = greenTime + yellowTime + redTime;
982 
983  if (counter >= totalCycle) {
984  counter = 0;
985  }
986 
987  // North-South signal states
988  if (counter < greenTime) {
989  NS_Signal = Green;
990  WE_Signal = Red;
991  } else if (counter < greenTime + yellowTime) {
992  NS_Signal = Yellow;
993  WE_Signal = Red;
994  } else {
995  NS_Signal = Red;
996  WE_Signal = Green;
997 
998  if (counter > totalCycle - yellowTime) {
999  WE_Signal = Yellow;
1000  }
1001  }
1002 }
